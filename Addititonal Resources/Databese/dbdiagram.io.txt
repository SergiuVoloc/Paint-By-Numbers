Table users as U {
  id  UUID [pk, increment]
  email varchar [unique]
  full_name varchar
  date_of_birth date
  phone varchar(30)
  password varchar
  created_at timestamp
  deleted boolean
}

Table address as A {
   id UUID [pk, increment]
   user_id UUID [ref: > U.id]
   full_name varchar
   address1 varchar
   address2 varchar
   postcode varchar
   phone varchar(30)
}

Table basket_item {
  id UUID [pk, increment]
  user_id UUID [ref: > U.id]
  product_details_id UUID [ref: > product_details.id]
  saved_for_later boolean
  quantity int
  time_added datetime
}

Table orders as O {
  id UUID [pk, increment]
  user_id UUID [ref: > U.id]
  address_id UUID [ref: > A.id]
  discount_id UUID [ref: > D.id]
  status_id UUID [ref: > order_status.id]
  created_at datetime
  modified datetime
  shipping_cost float
  amount float
  order_note varchar
}

Table order_status {
   id UUID [pk, increment]
   name varchar
}

Table order_item as OI {
   id UUID [pk, increment]
   order_id UUID [ref: > O.id]
   product_details_id UUID [ref: > product_details.id]
   quantity int
}

Table discounts as D {
  id UUID [pk, increment]
  name varchar
  discount float
  created_at datetime
  valid datetime
}

Table products as P {
  id UUID [pk, increment]
  name varchar
  description varchar
  price float
  category_id UUID [ref: > PC.id]
}

Table product_details {
  id UUID [pk, increment]
  size varchar
  product_id UUID [ref: > P.id]
}

Table categories as PC {
  id UUID [pk, increment]
  name varchar
}

Table tags {
  id UUID [pk, increment]
  product_id UUID [ref: > P.id]
  name varchar
}

Table product_photos {
  id UUID [pk, increment]
  product_id UUID [ref: > P.id]
  name varchar
  photo blob
}


Table file_storage {
   id UUID [pk, increment]
   product_id UUID [ref: > P.id]
   name varchar
   value varchar
}


Table product_details_attribute_values {
  id UUID [pk, increment]
  product_details_id UUID [ref: > product_details.id]
  attribute_values_id UUID [ref: > attribute_values.id]
}





// The simplest implementation of EAV may have just three tables: entity, attribute and value.
Table entities {
  id UUID [pk, increment]
  name varchar
}

Table attributes {
  id UUID [pk, increment]
  name varchar
}

Table attribute_values {
  id UUID [pk, increment]
  attribute_id UUID [ref: > attributes.id]
  entity_id UUID [ref: > entities.id]
  value varchar
}
